107void IRAM_ATTR handleInterrupt()
108{
109 portENTER_CRITICAL_ISR(&mux);
110 uint64_t TempVal = timerRead(timer); // value of timer at interrupt
111 PeriodCount = TempVal - StartValue; // period count between rising edges in 0.000001 of a second
112 StartValue = TempVal; // puts latest reading as start for next calculation
113 portEXIT_CRITICAL_ISR(&mux);
114 Last_int_time = millis();
115}

313// This task runs isolated on core 0 because sensors.requestTemperatures() is slow and blocking for about 750 ms
314void GetTemperature( void * parameter) {
315 float tmp0 = 0;
316 float tmp1 = 0;
317 for (;;) {
318 sensors.requestTemperatures(); // Send the command to get temperatures
319 vTaskDelay(100);
320 tmp0 = sensors.getTempCByIndex(0) + fTemp1Offset;
321 if (tmp0 != -127) OilTemp = tmp0;
322 vTaskDelay(100);
323 tmp1 = sensors.getTempCByIndex(1) + fTemp2Offset;
324 if (tmp1 != -127) MotTemp = tmp1;
325 vTaskDelay(100);
326 }
327}

329// Calculate engine RPM from number of interupts per time
330double ReadRPM() {
331 double RPM = 0;
332
333 portENTER_CRITICAL(&mux);
334 if (PeriodCount != 0) { // 0 means no signals measured
335 RPM = 1000000.00 / PeriodCount; // PeriodCount in 0.000001 of a second
336 }
337 portEXIT_CRITICAL(&mux);
338 if (millis() > Last_int_time + 200) RPM = 0; // No signals RPM=0;
339 return (RPM);
340}

445// ReadVoltage is used to improve the linearity of the ESP32 ADC see: https://github.com/G6EJD/ESP32-ADC-Accuracy-Improvement-function
446
447double ReadVoltage(byte pin) {
448 double reading = analogRead(pin); // Reference voltage is 3v3 so maximum reading is 3v3 = 4095 in range 0 to 4095
449 if (reading < 1 || reading > 4095) return 0;
450 // return -0.000000000009824 * pow(reading,3) + 0.000000016557283 * pow(reading,2) + 0.000854596860691 * reading + 0.065440348345433;
451 return (-0.000000000000016 * pow(reading, 4) + 0.000000000118171 * pow(reading, 3) - 0.000000301211691 * pow(reading, 2) + 0.001109019271794 * reading + 0.034143524634089) * 1000;
452} // Added an improved polynomial, use either, comment out as required
