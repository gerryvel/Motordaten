220 Serial.println("Scanning...");
221 nDevices = 0;
222 for(address = 1; address < 127; address++ )
223 {
224 Wire.beginTransmission(address);
225 error = Wire.endTransmission();
226 if (error == 0)
227 {
228 Serial.print("I2C device found at address 0x");
229 if (address<16)
230 {
231 Serial.print("0");
232 }
233 Serial.println(address,HEX);
234 nDevices++;
235 }
236 else if (error==4)
237 {
238 Serial.print("Unknow error at address 0x");
239 if (address<16)
240 {
241 Serial.print("0");
242 }
243 Serial.println(address,HEX);
244 nDevices++;
245 }
246 else if (error==4) {
247 Serial.print("Unknow error at address 0x");
248 if (address<16) {
249 Serial.print("0");
250 }
251 Serial.println(address,HEX);
252 }
253 }
254 if (nDevices == 0) {
255 Serial.println("No I2C devices found\n");
256 }
257 else {
258 Serial.println("done\n");
259 }
